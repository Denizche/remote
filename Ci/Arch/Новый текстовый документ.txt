#include <vcl.h>
#include <iostream.h>
#include <fstream.h>
#pragma hdrstop

//---------------------------------------------------------------------------

#pragma argsused
int main()
{
	string s="rrhutuygfes  ";                       // Данная строка
	int i,j;                                      // Ввод данных для счётчикоы
	char a[256];                                  // Массив символов
	int st[256];                                  // Массив чисел
	for (i = 0; i < 256; i++)
		{
			st[i]=0;
		}
			for (i = 0; i < 256; i++)             // Заполнение массива "а" символами ASCI таблицы
				{
					a[i]=i;
				}
				for (i = 0; i < s.length() ; i++) // Сбор статистики
					{
						for ( j=0 ; j < 256; j++)
						{
							if(a[j]==s[i])
							st[j]++;
						}
					}

	for (i = 0; i < 256; i++)
		{
			for (j = 0; j < 256; j++)
				{
					if (st[i]>st[j])              // Сортировка в порядке убывания
						{
							int k = st[i];
							char l=a[i];
							a[i]= a[j];
							st[i]=st[j];
							a[j]=l;
							st[j]=k;

						}
				}
		}
	for (i = 0; i < 256; i++)                     // Отсечение не встречающихся символов
		{
			if (st[i]!=0)
				{
					cout << a[i] << " : " << st[i] << endl;
				}
		}

	cin.get();
	return 0;
}
