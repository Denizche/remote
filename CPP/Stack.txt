struct Stack                   // Объявление параметров стэка
{                         // Начало объявления Стэка
string data;			  //Строка данных в стэке
int chnum;                 //Номер последнего элемента в стэке
};                         //Конец объявления Стэка
void add ( Stack& st, char ch  )     //Функция добавления эл-та в стэк
{                                 // Начало добавления эл-та в стэк
st.data[st.chnum] = ch;          // В позицию chnum в строку data записать элемент ch
st.chnum++;                   //перейти на след. позицию для эл-та.
}                           // Конец добавления эл-та в стэк
							 //Ничего нет. Пустая строка.
char out ( Stack& st )          //Ф-я вывода эл-та из стэка
{                             //Начало вывода эл-та из стэка
return st.data[--(st.chnum)];   //Вернуть последний символ
}                            //Конец вывода эл-та из стэка
							 //Ничего нет. Пустая строка.
void StFill ( Stack& st  )       //Заполнение стэка
{                                //Начало Заполнения стэка
 cout << "Enter chars " << "\n";   // Вывод строки "введите символ"
 char input;                    // объявление переменной для введенного символа
 while (input!='~')             // "Пока не конец строки"
 {                            //начало While
 cin >> input;                //считать с консоли символ
 if (input=='~')               //если это конец строки
  {                         //начало IF
   break;                    //Выйти из while
 }                         //Конец If
 add( st, input); // Помещение в стек элемента
 }                           //Конец while
}                           //Конец заполнения стэка
void StOut ( Stack& st  )       //  Вывод стэка
{                            //  Начало ввода стэка
 while (st.chnum>=0)        //Пока стек не кончится
 {                        // Начало While
  cout << out(st);         //Вывести элемент
 }                        // Конец While
}                       // Конец вывода стэка
void main()           // Ф-я main
{                    // Начало main
Stack st;             // объявление стэка ST
